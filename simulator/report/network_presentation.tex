% Slide 2
\slide{Scenario: The Digital Hospital}{
    Modern hospitals are digital ecosystems.
    \vspace{1cm}
    \begin{itemize}
        \item \textbf{Radiology}: Sending huge MRI images (Gigabytes).
        \item \textbf{Surgeons}: Video calling colleagues during operations.
        \item \textbf{Patients}: Using Guest Wi-Fi.
    \end{itemize}
    \vspace{1cm}
    \textbf{The Challenge:} How do we keep the surgeon's video smooth when a patient starts downloading a movie?
}

% Slide 3
\slide{What is Private SD-WAN?}{
    \textbf{Traditional WAN vs Private SD-WAN}
    \vspace{1cm}
    \begin{itemize}
        \item \textbf{Traditional}: Like a train on fixed tracks. Can't change route if there's a jam.
        \item \textbf{Private SD-WAN}: A dedicated network infrastructure (not shared with public internet). It sees traffic in real-time and steers data to the fastest open road---all within a secure, isolated environment.
    \end{itemize}
}

% Slide 4
\slide{The Danger: Congestion vs. DDoS}{
    \begin{itemize}
        \item \textbf{1. Legitimate Congestion}:
        \begin{itemize}
            \item \textbf{Cause}: Too much good work happening (e.g., 3 Doctors opening X-Rays).
            \item \textbf{Action}: \textbf{Queue it. Prioritize it.}
        \end{itemize}
        \vspace{0.5cm}
        \item \textbf{2. DDoS Attack}:
        \begin{itemize}
            \item \textbf{Cause}: Malicious Hackers / Botnets.
            \item \textbf{Action}: \textbf{Block it immediately.}
        \end{itemize}
    \end{itemize}
    \vspace{1cm}
    \textit{Misidentifying them is dangerous!}
}

% Slide 6
\slide{Solution: Weighted Fair Queuing (WFQ)}{
    \textbf{Imagine a Club Entrance with 3 Lines:}
    \vspace{1cm}
    \begin{itemize}
        \item \textbf{Gold Line (VIP)}: Lets 6 people in.
        \item \textbf{Silver Line}: Lets 3 people in.
        \item \textbf{Bronze Line}: Lets 1 person in.
    \end{itemize}
    \vspace{1cm}
    Even if the Bronze line has 1000 people, the VIPs still get through quickly.
}

        \item Real-time analysis gives us visibility.
    \end{enumerate}
